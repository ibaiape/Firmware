find_package(ament_cmake REQUIRED)
find_package(rclcpp REQUIRED)
find_package(rmw REQUIRED)
find_package(rmw_implementation_cmake REQUIRED)
find_package(std_msgs REQUIRED)

include_directories(
  include
  ${rclcpp_INCLUDE_DIRS}
)

px4_add_module(
	MODULE platforms__posix__drivers__barosim_ros2
	MAIN barosim_ros2
	COMPILE_FLAGS
		-std=c++11 -Wall -Wextra
	SRCS
		barosim_ros2.cpp
	DEPENDS
		platforms__common
		${rclcpp_LIBRARIES}
	)

ament_target_dependencies(platforms__posix__drivers__barosim_ros2 "rclcpp" "std_msgs")
